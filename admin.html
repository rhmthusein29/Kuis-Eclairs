async function nextQuestion() { document.getElementById("btn-next").disabled = true; let nextQ = currentQ + 1; document.getElementById("tag-status").innerText = "LOADING..."; try { let response = await fetch(`${WEB_APP_URL}?action=nextQuestion&q=${nextQ}`); let data = await response.json(); if(data.status === "success") { currentQ = nextQ; document.getElementById("active-num").innerText = currentQ; document.getElementById("text-pertanyaan").innerText = data.questionText; document.getElementById("tag-status").innerText = "SOAL " + currentQ; let timeLeft = 10; document.getElementById("admin-timer").innerText = timeLeft; clearInterval(timerInterval); timerInterval = setInterval(() => { timeLeft--; if (timeLeft < 0) timeLeft = 0; document.getElementById("admin-timer").innerText = timeLeft; if (timeLeft <= 0) { clearInterval(timerInterval); document.getElementById("btn-next").disabled = false; document.getElementById("tag-status").innerText = "WAKTU HABIS"; fetchResults(); } }, 1000); } } catch (error) { alert("Gagal memajukan pertanyaan!"); document.getElementById("btn-next").disabled = false; document.getElementById("tag-status").innerText = "ERROR"; } }
